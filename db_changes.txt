## ======================== Student Wallet Monetization System ========================
## Run all sections in order.
## Each section is independent — safe to run section by section.


## ── SECTION 1: Student Wallets ──────────────────────────────────────────────────────
## Core wallet table. One row per student. Tracks available, hold, and consumed balances
## separately so the UI can show all three without extra queries.

CREATE TABLE student_wallets (
    id                      BIGINT  AUTO_INCREMENT PRIMARY KEY,
    user_id                 BIGINT  NOT NULL,
    available_balance       DECIMAL(10,2)   NOT NULL DEFAULT 0.00,
    hold_balance            DECIMAL(10,2)   NOT NULL DEFAULT 0.00,
    consumed_balance        DECIMAL(10,2)   NOT NULL DEFAULT 0.00,
    free_applications_used  INT             NOT NULL DEFAULT 0,
    free_applications_limit INT             NOT NULL DEFAULT 3,
    created_at              TIMESTAMP       NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at              TIMESTAMP       NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,

    UNIQUE KEY uq_user_id (user_id),
    INDEX idx_user_id (user_id)
);


## ── SECTION 2: Wallet Transaction Ledger ────────────────────────────────────────────
## Permanent, append-only ledger. Every balance movement creates one row.
## Never update or delete rows — this is the source of truth for auditing.
##
## type values:
##   credit  – wallet recharged (online or manual approval)
##   hold    – amount reserved when student applies for a job
##   release – hold returned to available (rejection / auto-expiry)
##   consume – hold finalised as spent (interview confirmed/accepted)
##   refund  – exceptional manual refund by admin

CREATE TABLE wallet_transactions (
    id               BIGINT  AUTO_INCREMENT PRIMARY KEY,
    user_id          BIGINT  NOT NULL,
    amount           DECIMAL(10,2)   NOT NULL,
    type             ENUM('credit','hold','release','consume','refund') NOT NULL,
    reference_type   VARCHAR(50)     NULL COMMENT 'recharge | application',
    reference_id     BIGINT  NULL COMMENT 'wallet_recharges.id or application_holds.id',
    job_id           BIGINT  NULL,
    application_id   BIGINT  NULL,
    description      VARCHAR(255)    NULL,
    balance_before   DECIMAL(10,2)   NULL COMMENT 'available_balance before this operation',
    balance_after    DECIMAL(10,2)   NULL COMMENT 'available_balance after this operation',
    created_at       TIMESTAMP       NULL DEFAULT CURRENT_TIMESTAMP,

    INDEX idx_user_id       (user_id),
    INDEX idx_type          (type),
    INDEX idx_ref           (reference_type, reference_id),
    INDEX idx_application   (application_id),
    INDEX idx_job           (job_id)
);


## ── SECTION 3: Wallet Recharge Packs ────────────────────────────────────────────────
## Admin-configurable recharge amounts shown on the recharge screen.
## Adding a new pack = inserting a row; no code change needed.

CREATE TABLE wallet_recharge_packs (
    id         BIGINT  AUTO_INCREMENT PRIMARY KEY,

    label      VARCHAR(100)  NOT NULL,
    is_active  TINYINT(1)    NOT NULL DEFAULT 1,
    sort_order INT           NOT NULL DEFAULT 0,
    created_at TIMESTAMP     NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP     NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,

    INDEX idx_active_sort (is_active, sort_order)
);

## ── SECTION 4: Wallet Recharges ─────────────────────────────────────────────────────
## Mirrors firm_subscriptions / premium_requests pattern.
## Handles both Razorpay (online) and manual payment flows in one table.
## payment_method keeps gateway logic decoupled from wallet credit logic.
##
## status values:
##   pending  – awaiting payment or admin approval
##   approved – wallet has been credited
##   rejected – admin rejected manual payment proof
##   expired  – online payment order expired without completion

CREATE TABLE wallet_recharges (
    id                  BIGINT  AUTO_INCREMENT PRIMARY KEY,
    user_id             BIGINT  NOT NULL,
    amount              DECIMAL(10,2)   NOT NULL,
    payment_method      ENUM('razorpay','manual') NOT NULL DEFAULT 'razorpay',
    status              ENUM('pending','approved','rejected','expired') NOT NULL DEFAULT 'pending',

    -- Online payment (Razorpay / any future gateway)
    gateway_order_id    VARCHAR(255) NULL,
    gateway_payment_id  VARCHAR(255) NULL,
    gateway_signature   TEXT         NULL,
    payment_status      ENUM('pending','paid','failed') NULL DEFAULT 'pending',
    razorpay_response   JSON         NULL,
    payment_method_used VARCHAR(100) NULL COMMENT 'upi, card, netbanking etc.',

    -- Manual payment proof
    utr_number          VARCHAR(100) NULL,
    reference_number    VARCHAR(100) NULL,
    screenshot_url      TEXT         NULL,
    payment_date        DATE         NULL,

    -- Admin processing
    remarks             TEXT         NULL,
    approved_by         BIGINT  NULL,
    approved_at         TIMESTAMP    NULL,
    rejected_at         TIMESTAMP    NULL,

    -- Ledger link – set after wallet is credited
    wallet_transaction_id BIGINT  NULL,

    created_at          TIMESTAMP    NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at          TIMESTAMP    NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,

    INDEX idx_user_id           (user_id),
    INDEX idx_status            (status),
    INDEX idx_payment_method    (payment_method),
    INDEX idx_gateway_order     (gateway_order_id),
    INDEX idx_gateway_payment   (gateway_payment_id)
);


## ── SECTION 5: Application Holds ────────────────────────────────────────────────────
## One row per paid application (free applications do not create a hold row).
## Tracks the full lifecycle: held → consumed | released | expired.
##
## status values:
##   held      – amount reserved; application in progress
##   consumed  – interview confirmed/accepted; amount spent
##   released  – application rejected; amount returned to available
##   expired   – no recruiter action within 10 days; amount returned

CREATE TABLE application_holds (
    id                    BIGINT  AUTO_INCREMENT PRIMARY KEY,
    user_id               BIGINT  NOT NULL,
    application_id        BIGINT  NOT NULL,
    job_id                BIGINT  NOT NULL,
    amount                DECIMAL(10,2)   NOT NULL DEFAULT 49.00,
    status                ENUM('held','consumed','released','expired') NOT NULL DEFAULT 'held',
    held_at               TIMESTAMP       NULL DEFAULT CURRENT_TIMESTAMP,
    expires_at            TIMESTAMP       NULL COMMENT 'held_at + 10 days; used by auto-expiry job',
    consumed_at           TIMESTAMP       NULL,
    released_at           TIMESTAMP       NULL,
    release_reason        VARCHAR(100)    NULL COMMENT 'rejected | auto_expired | refund',

    -- Ledger links
    hold_transaction_id   BIGINT  NULL COMMENT 'wallet_transactions.id for the hold entry',
    settle_transaction_id BIGINT  NULL COMMENT 'wallet_transactions.id for consume/release/expire',

    created_at            TIMESTAMP       NULL DEFAULT CURRENT_TIMESTAMP,
    updated_at            TIMESTAMP       NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,

    UNIQUE KEY uq_application (application_id),
    INDEX idx_user_id     (user_id),
    INDEX idx_status      (status),
    INDEX idx_expires_at  (expires_at, status) COMMENT 'used by auto-expiry scheduler'
);

ADD COLUMN application_fee     DECIMAL(10,2)   NOT NULL DEFAULT 0.00  AFTER is_free_application,
ADD COLUMN wallet_hold_id      BIGINT  NULL                    AFTER application_fee;

-- Index for hold lookups
ALTER TABLE applications
ADD INDEX idx_wallet_hold (wallet_hold_id);


## ── SECTION 7: Backfill — Wallets for Existing Students ─────────────────────────────
## Creates wallet rows for all existing students so they start with the
## 3 free application quota. Run once after SECTION 1 is executed.
## New students get their wallet created at registration time (in code).

INSERT INTO student_wallets (user_id, available_balance, hold_balance, consumed_balance,
                             free_applications_used, free_applications_limit,
                             created_at, updated_at)
SELECT u.id, 0.00, 0.00, 0.00, 0, 3, NOW(), NOW()
FROM users u
WHERE u.role = 'student'
  AND u.is_deleted = 0
  AND NOT EXISTS (
      SELECT 1 FROM student_wallets sw WHERE sw.user_id = u.id
  );

-- Backfill free_applications_used for students who have already applied
-- (so existing applicants don't get a surprise extra free quota)
UPDATE student_wallets sw
JOIN (
    SELECT student_id, COUNT(*) AS total_apps
    FROM applications
    GROUP BY student_id
) app_counts ON app_counts.student_id = sw.user_id
SET sw.free_applications_used = LEAST(app_counts.total_apps, 3);

--  Phase 3 – Active Sessions & Login History  (2026-06-08)

-- user_sessions: one row per active login session.

-- expires_at mirrors token_expires_at so sessions are self-contained.
-- device_type: 'desktop' | 'mobile' | 'tablet'

    `id`               BIGINT  NOT NULL AUTO_INCREMENT,
    `user_id`          BIGINT  NOT NULL,
    `token`            VARCHAR(80)     NOT NULL,

    `ip_address`       VARCHAR(45)     NULL,

    `created_at`       TIMESTAMP       NULL,
    `updated_at`       TIMESTAMP       NULL,
    PRIMARY KEY (`id`),
    UNIQUE KEY `uq_user_sessions_token`    (`token`),
    KEY       `idx_user_sessions_user_id`  (`user_id`),
    CONSTRAINT `fk_user_sessions_user_id`
        FOREIGN KEY (`user_id`) REFERENCES `users` (`id`) ON DELETE CASCADE
);

-- login_history: append-only audit log; rows are never updated.

-- ============================================================
-- 2026-06-13 — Referral Rewards + SYS Coins
-- ============================================================
-- Adds a SYS Coin points currency (kept STRICTLY separate from wallet money) using
-- the same available/hold/consumed + ledger + holds pattern as the wallet, plus a
-- real-money referral payout ledger for referred firms that purchase premium.
-- No Eloquent migration — apply this SQL directly (project convention; code uses DB::table).
--
-- Coins are integers. Application coin cost = 50 (a deliberate round value, distinct
-- from the ₹49 money fee). Reward amounts live as constants in app/Helpers/SysCoinHelper.php
-- (WELCOME_BONUS=100, STUDENT_REFERRAL_BONUS=10, APPLICATION_COST=50, HOLD_DAYS=10) and
-- app/Helpers/ReferralHelper.php (FIRM_REFERRAL_REWARD=2000.00).

-- 1) Coin balance per user (mirrors student_wallets)
CREATE TABLE `sys_coin_accounts` (
  `id`              BIGINT  NOT NULL AUTO_INCREMENT,
  `user_id`         BIGINT  NOT NULL,
  `available_coins` INT NOT NULL DEFAULT 0,
  `hold_coins`      INT NOT NULL DEFAULT 0,
  `consumed_coins`  INT NOT NULL DEFAULT 0,
  `lifetime_earned` INT NOT NULL DEFAULT 0,
  `created_at`      TIMESTAMP NULL DEFAULT NULL,
  `updated_at`      TIMESTAMP NULL DEFAULT NULL,
  PRIMARY KEY (`id`),
  UNIQUE KEY `uniq_sys_coin_user` (`user_id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

-- 2) Coin ledger (mirrors wallet_transactions). type ∈
--    WELCOME_BONUS, REFERRAL_BONUS, APPLICATION_HOLD, APPLICATION_RELEASE,
--    APPLICATION_CONSUMED, ADMIN_CREDIT, BLOG_REWARD
CREATE TABLE `sys_coin_transactions` (
  `id`             BIGINT  NOT NULL AUTO_INCREMENT,
  `user_id`        BIGINT  NOT NULL,
  `amount`         INT NOT NULL,
  `type`           VARCHAR(40) NOT NULL,
  `reference_type` VARCHAR(40) NULL DEFAULT NULL,
  `reference_id`   BIGINT  NULL DEFAULT NULL,
  `application_id` BIGINT  NULL DEFAULT NULL,
  `job_id`         BIGINT  NULL DEFAULT NULL,
  `description`    VARCHAR(255) NULL DEFAULT NULL,
  `balance_before` INT NOT NULL DEFAULT 0,
  `balance_after`  INT NOT NULL DEFAULT 0,
  `created_at`     TIMESTAMP NULL DEFAULT NULL,
  PRIMARY KEY (`id`),
  KEY `idx_sys_coin_tx_user` (`user_id`),
  KEY `idx_sys_coin_tx_type` (`type`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

-- 3) Coin holds per application (mirrors application_holds). status ∈
--    held, consumed, released, expired
CREATE TABLE `sys_coin_holds` (
  `id`                    BIGINT  NOT NULL AUTO_INCREMENT,
  `user_id`               BIGINT  NOT NULL,
  `application_id`        BIGINT  NOT NULL,
  `job_id`                BIGINT  NULL DEFAULT NULL,
  `amount`                INT NOT NULL,
  `status`                VARCHAR(20) NOT NULL DEFAULT 'held',
  `held_at`               TIMESTAMP NULL DEFAULT NULL,
  `expires_at`            TIMESTAMP NULL DEFAULT NULL,
  `consumed_at`           TIMESTAMP NULL DEFAULT NULL,
  `released_at`           TIMESTAMP NULL DEFAULT NULL,
  `release_reason`        VARCHAR(40) NULL DEFAULT NULL,
  `hold_transaction_id`   BIGINT  NULL DEFAULT NULL,
  `settle_transaction_id` BIGINT  NULL DEFAULT NULL,
  `created_at`            TIMESTAMP NULL DEFAULT NULL,
  `updated_at`            TIMESTAMP NULL DEFAULT NULL,
  PRIMARY KEY (`id`),
  KEY `idx_sys_coin_hold_application` (`application_id`),
  KEY `idx_sys_coin_hold_user_status` (`user_id`, `status`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

-- 4) Real-money referral payouts for referred firms (mark-only; admin settles
--    externally — no wallet credit). status ∈ pending, approved, paid.
--    UNIQUE(referred_user_id) => one payout per referred firm (idempotency).
CREATE TABLE `referral_payouts` (
  `id`                   BIGINT  NOT NULL AUTO_INCREMENT,
  `referrer_user_id`     BIGINT  NOT NULL,
  `referred_user_id`     BIGINT  NOT NULL,
  `firm_subscription_id` BIGINT  NULL DEFAULT NULL,
  `plan`                 VARCHAR(40) NULL DEFAULT NULL,
  `reward_amount`        DECIMAL(10,2) NOT NULL DEFAULT 2000.00,
  `status`               VARCHAR(20) NOT NULL DEFAULT 'pending',
  `approved_by`          BIGINT  NULL DEFAULT NULL,
  `approved_at`          TIMESTAMP NULL DEFAULT NULL,
  `paid_at`              TIMESTAMP NULL DEFAULT NULL,
  `paid_reference`       VARCHAR(120) NULL DEFAULT NULL,
  `remarks`              VARCHAR(255) NULL DEFAULT NULL,
  `created_at`           TIMESTAMP NULL DEFAULT NULL,
  `updated_at`           TIMESTAMP NULL DEFAULT NULL,
  PRIMARY KEY (`id`),
  UNIQUE KEY `uniq_referral_payout_referred` (`referred_user_id`),
  KEY `idx_referral_payout_referrer` (`referrer_user_id`),
  KEY `idx_referral_payout_status` (`status`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_unicode_ci;

-- Messaging Phase 2 — denormalize conversations for unread-count + list scalability.
-- Adds last-message snapshot + per-side unread counters so the unread badge and the
-- conversation list never scan the messages table. Maintained in code on send/read.
ALTER TABLE conversations
  ADD COLUMN last_message_id          BIGINT NULL                   AFTER last_message_at,
  ADD COLUMN last_message_preview     VARCHAR(255) NULL             AFTER last_message_id,
  ADD COLUMN last_message_sender_type ENUM('candidate','firm') NULL AFTER last_message_preview,
  ADD COLUMN firm_unread_count        INT NOT NULL DEFAULT 0,
  ADD COLUMN candidate_unread_count   INT NOT NULL DEFAULT 0;

-- Backfill: last-message snapshot (latest message id per conversation).
UPDATE conversations c
JOIN (
  SELECT m.conversation_id, m.id, m.message, m.sender_type
  FROM messages m
  JOIN (SELECT conversation_id, MAX(id) AS max_id FROM messages GROUP BY conversation_id) x
    ON x.conversation_id = m.conversation_id AND x.max_id = m.id
) lm ON lm.conversation_id = c.id
SET c.last_message_id          = lm.id,
    c.last_message_preview     = LEFT(lm.message, 255),
    c.last_message_sender_type = lm.sender_type;

-- Backfill: per-side unread counters.
--   firm_unread_count      = unread messages sent BY candidate (firm hasn't read)
--   candidate_unread_count = unread messages sent BY firm (candidate hasn't read)
UPDATE conversations c SET
  firm_unread_count = (
    SELECT COUNT(*) FROM messages m
    WHERE m.conversation_id = c.id AND m.sender_type = 'candidate' AND m.is_read = 0
  ),
  candidate_unread_count = (
    SELECT COUNT(*) FROM messages m
    WHERE m.conversation_id = c.id AND m.sender_type = 'firm' AND m.is_read = 0
  );

-- =====================================================================
-- 2026-07-08 — Phase 2: Interview Credit Lifecycle (consume-on-confirm)
-- Business-rule change only. A free interview credit is now consumed ONLY
-- when the STUDENT confirms a scheduled interview (previously consumed when
-- the FIRM scheduled it). Applies to BOTH interview paths: the invite flow
-- (interview_invites) and the applications flow (applications). Reject and
-- timeout-expiry consume nothing (no refund needed — an unconfirmed
-- interview is simply never counted). Premium firms unchanged.
--
-- Consumption is tracked with an explicit `interview_credit_consumed_at`
-- timestamp (set once on confirm, never cleared) so it survives later
-- status changes (e.g. confirmed -> completed). FreeActionsHelper counts
-- distinct students with this set.
-- Applied to the local DB via tinker on 2026-07-08.
-- =====================================================================

-- interview_invites: credit marker, reschedule counter (max 1), new terminal
-- statuses for student reject + auto-expiry.
ALTER TABLE `interview_invites`
  ADD COLUMN `interview_credit_consumed_at` TIMESTAMP NULL AFTER `scheduled_at`,
  ADD COLUMN `reschedule_count` TINYINT UNSIGNED NOT NULL DEFAULT 0 AFTER `interview_credit_consumed_at`;

ALTER TABLE `interview_invites`
  MODIFY COLUMN `interview_status`
  ENUM('not_scheduled','scheduled','confirmed','completed','cancelled','rejected','expired')
  NOT NULL DEFAULT 'not_scheduled';

-- applications: same credit marker + reschedule counter for the applications
-- interview path. recruiter_status is free-text VARCHAR — expiry/reject use
-- new string values ('Interview Expired'), no enum change needed.
ALTER TABLE `applications`
  ADD COLUMN `interview_credit_consumed_at` TIMESTAMP NULL AFTER `interview_responded_at`,
  ADD COLUMN `interview_reschedule_count` TINYINT UNSIGNED NOT NULL DEFAULT 0 AFTER `interview_credit_consumed_at`;

-- Backfill so currently-consumed counts are preserved. Only interviews that
-- were CONFIRMED under the old flow keep a consumed credit; scheduled-but-
-- unconfirmed interviews are intentionally freed (new rule).
UPDATE `interview_invites`
  SET `interview_credit_consumed_at` = COALESCE(`scheduled_at`, `updated_at`, NOW())
  WHERE `interview_status` IN ('confirmed','completed')
    AND `interview_credit_consumed_at` IS NULL;

UPDATE `applications`
  SET `interview_credit_consumed_at` = COALESCE(`interview_responded_at`, `updated_at`, NOW())
  WHERE `recruiter_status` = 'Interview Confirmed'
    AND `interview_credit_consumed_at` IS NULL;

-- Admin-configurable confirmation timeout (system_settings; read via
-- SystemSettingService::getInterviewConfirmationTimeoutDays(), default 5).
INSERT INTO `system_settings`
  (`setting_key`, `setting_value`, `setting_type`, `title`, `description`, `category`, `is_editable`, `created_at`, `updated_at`)
SELECT * FROM (
  SELECT 'interview_confirmation_timeout_days' AS `setting_key`, '5' AS `setting_value`, 'integer' AS `setting_type`,
         'Interview Confirmation Timeout (days)' AS `title`,
         'Days a student has to confirm a scheduled interview before it auto-expires (no interview credit is consumed on expiry).' AS `description`,
         'interview' AS `category`, 1 AS `is_editable`, NOW() AS `created_at`, NOW() AS `updated_at`
) AS `seed`
WHERE NOT EXISTS (
  SELECT 1 FROM `system_settings` s WHERE s.`setting_key` = `seed`.`setting_key`
);

-- Additive only. A nullable, indexed interview_invite_id lets the Activity
-- Center (firm_activities) and the Notification bell (notifications) resolve
-- an invite's LIVE status on read and render the shared ScheduleInterviewDialog
-- CTA (accepted+not_scheduled) or the final "Interview Invitation Rejected"
-- state (declined/rejected) — without duplicating any interview state onto the
-- feed rows. Non-invite rows keep NULL and render exactly as before.
-- Applied via Laravel migration
-- 2026_07_09_000001_add_interview_invite_link_to_feeds. SQL equivalent below.
-- =====================================================================

ALTER TABLE `firm_activities`
  ADD COLUMN `interview_invite_id` BIGINT UNSIGNED NULL AFTER `action`,
  ADD KEY `firm_activities_interview_invite_id_index` (`interview_invite_id`);

ALTER TABLE `notifications`
  ADD COLUMN `interview_invite_id` BIGINT UNSIGNED NULL AFTER `action_url`,
  ADD KEY `notifications_interview_invite_id_index` (`interview_invite_id`);
